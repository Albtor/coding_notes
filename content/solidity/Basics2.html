    pragma solidity 0.5.1;

    contract MyContract {
        mapping(address = > uint256) public balances; //associative array
        address payable wallet;

        constructor(address payable _wallet) public {
            wallet = _wallet;
        }

        function buyToken() public payable{ //payable makes the function able to accept eth 
            //buy a buy Ethereum
            balances[msg.sender] += 1;
            //sent ether to the wallet
            wallets.transfer(msg.value); //msg.value tells you how much eth is sent
            
        }
    }

    //Solidity  INHERIT SMART CONTRACTS

    pragma solidity 0.5.1;

    contract ERC20Token{
        string public name;
        mapping(address = > uint256) public balances; //associative array

        constructor(string memory _name) public{
            name = _name;
        }
        function mint() public{
            balances[tx.origin] ++; //tx.origin always the person who originated the transaction. Tricky !
        }
    }


    contract MyToken is ERC20Token {
        //string public name = "My Token";
        string public symbol;
        address[] public owners;
        uint256 ownerCount;

        constructor(string memory _name, string memory _symbol) ERC20Token(_name) public {
            symbol = _symbol;
        }

        function mint() public{
            super.mint();
            ownerCount ++;
            owners.push(msg.sender);
        }
    }



    //Solidity  Common library SafeMath
    //libraries organize code that can be reused //DRY : DOn't repeat yourself
    

    library Math{
        function divide(uint256 a, uint256 b) internal pure returns(uint256) {
            require(b>0); //avoid problem if value 2 is 0
            uint c = a/b;
            return c;
        }
    }
    
    contract MyContract{
        uint 256 public value;

        function calculate(uint _value1, uint value2) public{
            value = Math.divide(_value1, _value2);
        }
    }

    //import from another file instead:
    import "./Math.sol";
