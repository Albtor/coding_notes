    //1- IF ELSE ELSE IF
    public void ifElse() {
        int time = 22;
        if (time < 10) {
            System.out.println("Good morning.");
        } else if (time < 20) {
            System.out.println("Good day.");
        } else {
            System.out.println("Good evening.");
        }
    }

    //SWITCH
    public void switching() {
        int day = 4;
        switch (day) {
            case 6:
                System.out.println("Today is Saturday");
                break;
            case 7:
                System.out.println("Today is Sunday");
                break;
            default:
                System.out.println("Looking forward to the Weekend");
        }

    }

    //WHILE
    public void Whiles() {
        int i = 0;
        while (i < 5) {
            System.out.println(i);
            i++;
        }
    }

    //DO WHILE
    public void doWhile() {
        int i = 0;
        do {
            System.out.println(i);
            i++;
        } while (i < 5);

    }

    //FOR
    public void forLoop() {
        for (int i = 0; i < 5; i++) {
            System.out.println(i);
        }

        String[] cars = {"Volvo", "BMW", "Ford", "Mazda"};
        for (String i : cars) {
            System.out.println(i);
        }
    }

    //BREAK // can be used to joup out of a loop
    public void breaking() {
        for (int i = 0; i < 10; i++) {
            if (i == 4) {
                break;
            }
            System.out.println(i);
        }
    }

    //CONTINUE
    public void continuing() {
        for (int i = 0; i < 10; i++) {
            if (i == 4) {
                continue;
            }
            System.out.println(i);
        }
    }
