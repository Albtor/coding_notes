    //CONNEXION Class Basic
    public class Conexion {
    public Connection con;
    
    public  Conexion(){
        con=null;
        try{
            Class.forName("com.mysql.cj.jdbc.Driver");
            con = DriverManager.getConnection("jdbc:mysql://localhost:3306/informaticaatr?useSSL=false&useTimezone=true&serverTimezone=UTC","xxxx","xxxx");
            if (con != null){
                System.out.println("Conectado a la BBDD");
            }
            
        }catch (Exception e){
            System.out.println("Error de conexión");
            System.err.println("Error:" + e);
        }
        }
        
    }
    



    // CONEXION DInterfaces
    private void mostrarDatos(Integer id) {
        Conexion cn = new Conexion();
        Statement st;
        ResultSet rs;
        try{
            st = cn.con.createStatement();
            rs = st.executeQuery("SELECT * from productosatr WHERE id = "+id+"");
            
            while (rs.next()){

                jTextFieldNombre.setText(rs.getString("nombre"));
                jTextFieldEspecificaciones.setText(rs.getString("especificaciones"));
                jTextFieldPrecio.setText(rs.getString("precio")+" €");
            }
            cn.con.close();
       }catch (SQLException e){
            System.out.println("Error");
        }

}
