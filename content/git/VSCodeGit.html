    // AUTHENTICATION
    password vs ssh

    git remote -v

    PASSWD - origin https://github.com/Albtor/3dsocialmediaicons.git (fetch) (push) 
    SSH - origin git@github.com:Albtor/3dsocialmediaicons.git (fetch) (push)
