    // SASS NESTING
    Be careful, it's better to use classes that can be reused

    .main {
        width: 80%;
        margin: 0 auto;
      
        .main__paragraph {
          font-weight: map-get($font-weights, bold);
        }
      }




    //& means the parent
    .main {
        width: 80%;
        margin: 0 auto;
      
        #{&}__paragraph {
          font-weight: map-get($font-weights, bold);
      
          &:hover {
            color: $accent-color;
          }
        }
      }
